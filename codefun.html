static void FakeCalculate()
{
    Console.Write(" Calculating");
    for (int i = 0; i < 6; i++)
    {
        Console.Write(".");
        Thread.Sleep(150);
        Console.Write(".");
        Thread.Sleep(150);
        Console.Write(".");
        Thread.Sleep(150);
        Console.Write("\b\b\b");
        Console.Write("   ");
        Console.Write("\b\b\b");
        Thread.Sleep(150);
    }
    Console.Write(" [Done]");
    Console.WriteLine();
}

C# Crazy Print Animation

                    
static void CrazyPrint(string stringToPrint)
{
    char[] characters = stringToPrint.ToCharArray();
    int stringLength = stringToPrint.Length;
    Random rnd = new Random();

    for (int i = 0; i < stringLength; i++)
    {
        int ranSleep = rnd.Next(20, 70);
        for (int j = 0; j < 10; j++)
        {
            Console.Write(RandomChar(rnd));
            Thread.Sleep(ranSleep);
            Console.Write("\b");
        }
        Console.Write($"{characters[i]}");
    }

    static char RandomChar(Random rnd)
    {
        int ranChar = rnd.Next(97, 123);
        return (char)ranChar;
    }
}
